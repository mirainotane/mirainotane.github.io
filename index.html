function showResult() {

  const q1 = document.getElementById('q1Answer').value.trim();

  const checked = document.querySelectorAll('#checkboxGrid input:checked');

  const total = ABILITIES.length; // 20

  const selectedCount = checked.length;

  const ratio = selectedCount / total;


  // タグ集計

  const tagCount = { en: 0, math: 0, jp: 0, sports: 0 };

  checked.forEach(cb => { tagCount[cb.value] = (tagCount[cb.value] || 0) + 1; });


  // スコア計算（科目ごとに重み付け）

  const scores = {

    en:     tagCount.en * 2,

    math:   tagCount.math * 2 + (tagCount.en > 0 ? 1 : 0),

    jp:     tagCount.jp * 2 + (tagCount.en > 0 ? 1 : 0),

    prog:   tagCount.math * 1.5 + tagCount.jp * 0.5,

    sports: tagCount.sports * 2,

  };


  // 上位3科目を選ぶ

  const sorted = Object.entries(scores).sort((a, b) => b[1] - a[1]);

  const top3 = sorted.slice(0, 3).map(([key]) => key);


  // 8割以上の場合は特別提案を表示

  const isSpecial = ratio >= 0.8;


  // 結果画面を構築

  document.getElementById('quizProgressFill').style.width = '100%';

  document.getElementById('quizStep2').classList.remove('active');

  document.getElementById('quizResult').classList.add('active');


  document.getElementById('resultQ1Summary').textContent = q1 || '（入力なし）';

  document.getElementById('resultQ2Count').innerHTML =

    `選択した能力：<span>${selectedCount}項目 / 20項目</span>${isSpecial ? '　→ 幅広い力をお望みです！' : ''}`;


  if (isSpecial) {

    document.getElementById('resultSpecial').classList.add('show');

  }


  // カード生成

function sendToGAS(q1, checked, top3, isSpecial, ratio) {

  if (!GAS_URL || GAS_URL === 'https://script.google.com/macros/s/AKfycbw_sr-a_LLXkik9a_yxk2o5utp13v_Rg950-9zkEGrl9Y-JvqNwdCQc26sSePRzopbLFQ/exec') return;

  const selectedLabels = Array.from(checked).map(cb => {

    const idx = parseInt(cb.id.replace('ab', ''));

    return ABILITIES[idx].label;

  });

  const payload = {

    timestamp: new Date().toLocaleString('ja-JP'),

    q1: q1,

    selectedAbilities: selectedLabels.join(' / '),

    selectedCount: checked.length,

    ratio: Math.round(ratio * 100) + '%',

    proposed: top3.map(k => SUBJECTS[k].name).join(' / '),

    isSpecial: isSpecial ? '金曜最終コース提案' : '通常提案',

  };

  fetch(GAS_URL, {

    method: 'POST',

    mode: 'no-cors',

    headers: { 'Content-Type': 'application/json' },

    body: JSON.stringify(payload),

  }).catch(() => {});

}
